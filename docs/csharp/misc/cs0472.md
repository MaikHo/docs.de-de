```csharp  
public class Test  
{  
    public static int Main()  
    {  
        int i = 5;  
        int counter = 0;  
  
        // Comparison:  
        if (i == null)  // CS0472  
        // To resolve, use a valid value for i.  
            counter++;
        return counter;  
    }  
}
```
